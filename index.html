        document.addEventListener("DOMContentLoaded", function () {

            const statusElement = document.getElementById("status");

            const entryForm = document.getElementById("entryForm");

            const classSelection = document.getElementById("classSelection");

            const cabImageField = document.getElementById("cabImageField");


            function showStatus(message, isError = false) {

                statusElement.textContent = message;

                statusElement.style.color = isError ? "red" : "green";

            }


            // LIFFの初期化

            liff.init({ liffId: "YOUR_LIFF_ID" })

                .then(() => {

                    console.log("LIFF initialized");

                    return liff.getProfile();

                })

                .then(profile => {

                    console.log("Profile:", profile);

                    document.getElementById("entryName").value = profile.displayName;

                })

                .catch(err => {

                    console.error("LIFF初期化エラー:", err);

                    showStatus("エラー: LIFFの初期化に失敗しました。ページを再読み込みしてください。", true);

                });


            // クラス選択による表示切替

            classSelection.addEventListener("change", function () {

                cabImageField.style.display = (this.value === "45") ? "block" : "none";

            });


            // フォーム送信処理

            entryForm.addEventListener("submit", async function (e) {

                e.preventDefault();

                showStatus("送信中...");


                const formData = new FormData(this);


                try {

                    const response = await fetch('YOUR_API_ENDPOINT', {

                        method: 'POST',

                        body: formData

                    });


                    if (!response.ok) {

                        throw new Error('サーバーエラーが発生しました');

                    }


                    const result = await response.json();

                    showStatus("エントリーが正常に送信されました！");

                    console.log(result);

                } catch (error) {

                    console.error('送信エラー:', error);

                    showStatus("エントリーの送信中にエラーが発生しました。もう一度お試しください。", true);

                }

            });

        });
