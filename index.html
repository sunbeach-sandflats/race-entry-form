        <label>チーム名:</label>

        <label>クラス選択:</label>

        <label>車体右画像:</label>

        <input type="file" id="imageRight" accept="image/*" required>


        <label>車体左画像:</label>

        <input type="file" id="imageLeft" accept="image/*" required>


        <!-- 45クラスのみキャブ画像 -->

        <div id="cabImageField" style="display:none;">

            <label>キャブ画像:</label>

    <p id="status"></p> <!-- ステータスメッセージ用 -->


    <!-- LIFFの初期化とフォーム処理 -->

    <script>

        document.addEventListener("DOMContentLoaded", function () {

            // LIFFの初期化

            liff.init({

                liffId: "2006687211-nxm8KDMb" // ここに発行されたLIFF IDを設定

            })

            .then(() => {

                console.log("LIFF initialized");

                return liff.getProfile(); // LINEプロフィールを取得

            })

            .then(profile => {

                console.log("Profile:", profile);

                // LINEの表示名をエントリーネーム欄にセット

                document.getElementById("entryName").value = profile.displayName;

            })

            .catch(err => {

                console.error("LIFF初期化エラー:", err);

                document.getElementById("status").innerText = "エラー: LIFFの初期化に失敗しました";

            });


            // クラス選択によるキャブ画像フィールドの表示切替

            document.getElementById("classSelection").addEventListener("change", function () {

                const selectedClass = this.value;

                document.getElementById("cabImageField").style.display = (selectedClass === "45") ? "block" : "none";

            });


            // フォーム送信処理

            document.getElementById("entryForm").addEventListener("submit", function (e) {

                e.preventDefault();


                // 入力データの取得

                const entryName = document.getElementById("entryName").value;

                const teamName = document.getElementById("teamName").value;

                const classSelection = document.getElementById("classSelection").value;

                const imageRight = document.getElementById("imageRight").files[0];

                const imageLeft = document.getElementById("imageLeft").files[0];

                const imageCab = (classSelection === "45") ? document.getElementById("imageCab").files[0] : null;


                // 確認用のコンソール出力

                console.log("エントリーネーム:", entryName);

                console.log("チーム名:", teamName);

                console.log("クラス:", classSelection);

                console.log("車体右画像:", imageRight);

                console.log("車体左画像:", imageLeft);

                if (imageCab) console.log("キャブ画像:", imageCab);


                // ユーザーへの確認

                alert("エントリーが送信されました！\n" +

                      "エントリーネーム: " + entryName + "\n" +

                      "チーム名: " + teamName + "\n" +

                      "クラス: " + classSelection);


                // ★ここでサーバー送信処理やGoogleスプレッドシートへの連携を追加する

            });

        });

    </script>
